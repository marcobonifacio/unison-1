-- | Needed for traversals.

-- | Needed for setters.
newtype FunArrow i a b = FunArrow { forall i a b. FunArrow i a b -> a -> b

newtype IxStar f i a b = IxStar { forall (f :: * -> *) i a b. IxStar f i a b -> i -> a -> f b
